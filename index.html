<body>

  <h1>🧠 反轉反應挑戰</h1>

  <p>看到「左」請按→，看到「右」請按←</p>

  <div id="prompt">準備好了嗎？</div>

  <div id="result"></div>

  <div id="score">得分：0</div>

  <button onclick="startGame()">開始遊戲</button>


  <script>

    let score = 0;

    let current = '';

    let timer;


    function showPrompt() {

      const options = ['左', '右'];

      current = options[Math.floor(Math.random() * 2)];

      document.getElementById('prompt').textContent = current;

      timer = setTimeout(() => {

        document.getElementById('result').textContent = '⏱️ 太慢了！';

        score = 0;

        updateScore();

      }, 2000);

    }


    function updateScore() {

      document.getElementById('score').textContent = `得分：${score}`;

    }


    function handleKey(e) {

      if (!current) return;

      clearTimeout(timer);

      const correct = (current === '左' && e.key === 'ArrowRight') || (current === '右' && e.key === 'ArrowLeft');

      document.getElementById('result').textContent = correct ? '✅ 正確！' : '❌ 錯誤！';

      score = correct ? score + 1 : 0;

      updateScore();

      setTimeout(() => {

        document.getElementById('result').textContent = '';

        showPrompt();

      }, 800);

    }


    function startGame() {

      score = 0;

      updateScore();

      document.getElementById('result').textContent = '';

      showPrompt();

    }


    document.addEventListener('keydown', handleKey);

  </script>

</body>
